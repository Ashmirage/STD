// 调度任务参数
typedef struct
{
	void (*func)(void); //任务函数
	uint32_t hz;        //执行频率
	uint16_t tick;      //执行间隔tick
	uint32_t last;      //上次执行时间
}task_t;

#define TASK_N  (sizeof(tasks) / sizeof(tasks[0]))

// 任务表
static task_t tasks[] = {
	{Loop_1000hz,1000,0,0},
	{Loop_50hz,50,0,0},
	{Loop_2hz,2,0,0},
	{Loop_1hz,1,0,0},
};

// 裸机任务调度器初始化
void Scheduler_init(void)
{
	uint8_t i;
	for(i = 0;i < TASK_N;i++)
	{
		tasks[i].tick = (uint16_t)1000 / tasks[i].hz; //1000 / 频率 = 执行间隔(ms)
		if(tasks[i].tick < 1) //最快也至少 1ms 执行一次
		{
			tasks[i].tick = 1;
		}
	}
}

// 主循环里面不断循环执行这个函数
void Scheduler_run(void)
{
	uint8_t i;
	uint32_t now = SysTick_GetTick(); // 一次读取，所有任务共用
	// 轮询检查不同周期的函数是否需要执行
	for(i = 0;i < TASK_N;i++)
	{
		if((now - tasks[i].last) >= tasks[i].tick)
		{
			tasks[i].last = now; //更新时间
			tasks[i].func();     //执行任务
		}
	}
}
void Hardware_init(void)
{
    delay_init(168);//时钟初始化
	uart_init(115200);//串口调试
	LCD_Init(); //LCD显示屏

	Motor_init(); //直流电机初始化
	RTC_clk_init();  //实时时钟初始化

	Lcd_bootup_scrolling();  //滚动字幕);化
}
