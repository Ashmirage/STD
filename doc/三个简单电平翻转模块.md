PF9,PF10分别是两个LED

PF8是蜂鸣器

```c
#include "sys.h"

/* PF8 高电平响，PF9/PF10 低电平亮。 */
void io0(void)
{
	GPIO_InitTypeDef g;

	RCC_AHB1PeriphClockCmd(RCC_AHB1Periph_GPIOF, ENABLE);

	g.GPIO_Pin = GPIO_Pin_8 | GPIO_Pin_9 | GPIO_Pin_10;
	g.GPIO_Mode = GPIO_Mode_OUT;
	g.GPIO_OType = GPIO_OType_PP;
	g.GPIO_Speed = GPIO_Speed_50MHz;
	g.GPIO_PuPd = GPIO_PuPd_NOPULL;
	GPIO_Init(GPIOF, &g);

	PFout(8) = 0;
	PFout(9) = 1;
	PFout(10) = 1;
}

void b1(void) { PFout(8) = 1; }
void b0(void) { PFout(8) = 0; }
void bt(void) { PFout(8) = !PFout(8); }

void r1(void) { PFout(9) = 0; }
void r0(void) { PFout(9) = 1; }
void rt(void) { PFout(9) = !PFout(9); }

void g1(void) { PFout(10) = 0; }
void g0(void) { PFout(10) = 1; }
void gt(void) { PFout(10) = !PFout(10); }

```
